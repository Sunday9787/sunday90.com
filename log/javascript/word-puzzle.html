  function aroundUnit(arry, x, y, selectRow) {
    let n = selectRow ? x : y;
    let len = arry.length;
    let unit = [];
    for (let i = 0; i < len; i++) {
      if (arry[n + i] && arry[n + i].isEmpty) break;
      if (arry[n + i]) {
        unit.push(arry[n + i]);
      }
    }

    for (let i = 1; i < len; i++) {
      if (arry[n - i] && arry[n - i].isEmpty) break;
      if (arry[n - i]) unit.push(arry[n - i]);
    }
    return unit;
  }
