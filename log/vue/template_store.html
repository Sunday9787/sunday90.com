   * @type {Template.Store}

  const state = reactive({
    name: '',
    direction: 'vertical',
    get changes() {

      changes = val
      this.componentsData = Array.from(this.components.values())

      /** @type {number} */
      get width() {

      get height() {
        return size[this.name][state.direction].height
